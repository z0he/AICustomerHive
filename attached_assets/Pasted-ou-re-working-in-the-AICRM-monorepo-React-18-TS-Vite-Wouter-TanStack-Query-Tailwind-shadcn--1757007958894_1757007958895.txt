C) Query param helpers (src/lib/useQueryParam.ts)

import { useLocation } from "wouter";

export function useQueryParams(): [URLSearchParams, (next: Record<string, string | undefined>) => void] {
  const [loc, setLoc] = useLocation();
  const url = new URL(loc, "http://dummy");
  const params = new URLSearchParams(url.search);
  const setParams = (next: Record<string, string | undefined>) => {
    Object.entries(next).forEach(([k, v]) => {
      if (v === undefined || v === null || v === "") params.delete(k);
      else params.set(k, String(v));
    });
    setLoc(url.pathname + "?" + params.toString());
  };
  return [params, setParams];
}


D) Contacts page (src/pages/contacts/ContactsPage.tsx)

import { useQuery } from "@tanstack/react-query";
import { useQueryParams } from "@/lib/useQueryParam";
import { Link } from "wouter";
import clsx from "clsx";
import React from "react";

const STAGES = [
  { key: "all",         label: "All" },
  { key: "lead",        label: "Leads" },
  { key: "mql",         label: "MQL" },
  { key: "opportunity", label: "Opportunities" },
  { key: "customer",    label: "Customers" },
  { key: "evangelist",  label: "Evangelists" },
  { key: "churned",     label: "Churned" },
] as const;

type StageKey = typeof STAGES[number]["key"];

export default function ContactsPage() {
  const [params, setParams] = useQueryParams();
  const stage = (params.get("stage") as StageKey) || "all";
  const q = params.get("q") || "";
  const owner = params.get("owner") || "";

  const { data, isLoading, isError } = useQuery({
    queryKey: ["contacts", { stage, q, owner }],
    queryFn: async () => {
      const usp = new URLSearchParams({ stage, q, owner });
      const res = await fetch(`/api/contacts?` + usp.toString());
      if (!res.ok) throw new Error("Failed to load contacts");
      return res.json() as Promise<{ rows: ContactRow[] }>;
    },
    staleTime: 30_000,
  });

  return (
    <div className="p-6">
      <h1 className="text-xl font-semibold mb-4">Contacts</h1>

      <StagePills active={stage} onChange={(s)=>setParams({ stage: s })} />

      <div className="mt-4 flex gap-2">
        <input
          className="input input-bordered w-80"
          placeholder="Search name, email…"
          defaultValue={q}
          onKeyDown={(e)=>{ if(e.key==="Enter") setParams({ q: (e.target as HTMLInputElement).value })}}
        />
        {/* Owner filter placeholder */}
        <select className="select select-bordered" defaultValue={owner}
          onChange={(e)=>setParams({ owner: e.target.value || undefined })}>
          <option value="">All owners</option>
          <option value="me">My contacts</option>
        </select>
        <button className="btn btn-ghost">Advanced filters</button>
        <div className="ml-auto flex gap-2">
          <button className="btn">Import</button>
          <button className="btn btn-primary">Create contact</button>
        </div>
      </div>

      <div className="mt-4">
        {isLoading && <TableSkeleton />}
        {isError && <div className="text-red-600">Failed to load contacts.</div>}
        {!isLoading && !isError && <ContactsTable rows={data?.rows || []} />}
      </div>
    </div>
  );
}

function StagePills({ active, onChange }:{ active: StageKey; onChange:(s:StageKey)=>void }) {
  return (
    <div className="flex flex-wrap gap-2">
      {STAGES.map(s => (
        <button key={s.key}
          onClick={()=>onChange(s.key)}
          className={clsx(
            "px-3 py-1 rounded-full border text-sm",
            active===s.key ? "bg-green-600 text-white border-green-600" : "bg-white hover:bg-gray-50"
          )}>
          {s.label}
        </button>
      ))}
    </div>
  );
}

type ContactRow = {
  id: string;
  firstName?: string; lastName?: string; email: string;
  jobTitle?: string; industry?: string; country?: string;
  lifecycleStage?: string; source?: string; lastActivityAt?: string;
  ownerName?: string;
};

function ContactsTable({ rows }: { rows: ContactRow[] }) {
  if (rows.length === 0) return <EmptyState />;
  return (
    <div className="overflow-auto rounded border">
      <table className="w-full text-sm">
        <thead className="bg-gray-50 text-gray-600">
          <tr>
            <Th>Name</Th><Th>Email</Th><Th>Job Title</Th><Th>Industry</Th>
            <Th>Country/Region</Th><Th>Lifecycle Stage</Th><Th>Source</Th>
            <Th>Last activity date</Th><Th>Owner</Th>
          </tr>
        </thead>
        <tbody>
          {rows.map(r => (
            <tr key={r.id} className="hover:bg-gray-50">
              <Td>
                <Link href={`/contacts/${r.id}`} className="text-green-700 hover:underline">
                  {(r.firstName || r.lastName) ? `${r.firstName ?? ""} ${r.lastName ?? ""}`.trim() : "—"}
                </Link>
              </Td>
              <Td>{r.email}</Td>
              <Td>{r.jobTitle ?? "—"}</Td>
              <Td>{r.industry ?? "—"}</Td>
              <Td>{r.country ?? "—"}</Td>
              <Td>{r.lifecycleStage ?? "—"}</Td>
              <Td>{r.source ?? "—"}</Td>
              <Td>{r.lastActivityAt ? new Date(r.lastActivityAt).toLocaleDateString() : "—"}</Td>
              <Td>{r.ownerName ?? "—"}</Td>
            </tr>
          ))}
        </tbody>
      </table>
    </div>
  );
}

function Th({ children }:{children:React.ReactNode}) {
  return <th className="text-left font-medium px-3 py-2 whitespace-nowrap">{children}</th>;
}
function Td({ children }:{children:React.ReactNode}) {
  return <td className="px-3 py-2 border-t">{children}</td>;
}
function TableSkeleton() {
  return <div className="animate-pulse h-40 rounded border bg-gray-50" />;
}
function EmptyState() {
  return (
    <div className="p-10 text-center text-gray-600 border rounded">
      No contacts match this view.
    </div>
  );
}
