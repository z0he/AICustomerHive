function Group({ label }: { label: string }) {
  return <div className="px-4 pt-6 pb-2 text-xs font-semibold text-slate-500">{label}</div>;
}

function Item({ item }: { item: NavItem }) {
  const Base = (
    <NavLink
      to={item.href}
      className={({ isActive }) =>
        `flex items-center gap-2 px-4 py-2 text-sm hover:bg-slate-50 ${
          isActive ? 'text-emerald-700 font-medium border-l-2 border-emerald-600 bg-emerald-50' : 'text-slate-700'
        }`
      }
      end
    >
      {item.icon ? <item.icon size={16} /> : null}
      <span>{item.label}</span>
    </NavLink>
  );
  if (!item.children?.length) return <li>{Base}</li>;
  return (
    <li>
      {Base}
      <ul className="ml-2 mt-1 border-l border-slate-200">
        {item.children.map((c) => (
          <li key={c.href}>
            <NavLink
              to={c.href}
              className={({ isActive }) =>
                `block py-1.5 pl-6 pr-3 text-[13px] hover:bg-slate-50 ${
                  isActive ? 'text-emerald-700 font-medium' : 'text-slate-600'
                }`
              }
              end
            >
              {c.label}
            </NavLink>
          </li>
        ))}
      </ul>
    </li>
  );
}

export function Sidebar() {
  return (
    <aside className="w-64 border-r border-slate-200 bg-white">
      <div className="h-14 flex items-center px-4 font-semibold"> {/* branding comes later */}
        AICRM
      </div>
      <nav className="py-2">
        {NAV.map((g) => (
          <div key={g.label}>
            <Group label={g.label} />
            <ul className="space-y-0.5">{g.items.map((i) => <Item key={i.href} item={i} />)}</ul>
          </div>
        ))}
      </nav>
    </aside>
  );
}
